安装RabbitMQ ​

临时安装emqx

shell
docker run -d --rm --name hangzhou-rabbitmq rabbitmq:latest
1

创建本地存储映射

shell
mkdir -p /docker/rabbitmq/etc/rabbitmq
1

复制容器文件至本地映射文件

shell
docker cp hangzhou-rabbitmq:/etc/rabbitmq/conf.d /docker/rabbitmq/etc/rabbitmq/
1
shell
docker run -d --restart=always --name hangzhou-rabbitmq -e RABBITMQ_DEFAULT_USER=otn -e RABBITMQ_DEFAULT_PASS='Otn2020!HZ' -v /docker/rabbitmq/etc/rabbitmq/conf.d:/etc/rabbitmq/conf.d -p 15672:15672 -p 5672:5672 rabbitmq:latest
1

进入容器

shell
docker exec -it hangzhou-rabbitmq  bash
1

开启管理端

shell
rabbitmq-plugins enable rabbitmq_management
1

安装EMQX ​

临时安装emqx

shell
docker run -d --name hangzhou-emqx --rm emqx/emqx
1

创建本地存储映射

shell
mkdir -p /docker/emqx/{bin,etc,lib,data,log}
1

复制容器文件至本地映射文件

shell
docker cp hangzhou-emqx:/opt/emqx/data /docker/emqx
docker cp hangzhou-emqx:/opt/emqx/etc /docker/emqx
docker cp hangzhou-emqx:/opt/emqx/lib /docker/emqx
docker cp hangzhou-emqx:/opt/emqx/log /docker/emqx
1
2
3
4

赋予本地映射文件权限

shell
chown -R 1000:1000 /docker/emqx/
chmod -R 777 /docker/emqx/
1
2

shell
docker run -d --name hangzhou-emqx --restart=always  --privileged=true -v /docker/emqx/data:/opt/emqx/data -v /docker/emqx/etc:/opt/emqx/etc -v /docker/emqx/log:/opt/emqx/log -v /docker/emqx/lib:/opt/emqx/lib -p 1883:1883 -p 8083:8083 -p 8084:8084 -p 8883:8883 -p 18083:18083 emqx/emqx
